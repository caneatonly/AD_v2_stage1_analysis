\section*{Nomenclature}

\subsection*{Latin Symbols}
\begin{description}
\item[$A_{ref}$] Reference area used in hydrodynamic coefficient scaling (m$^2$).
\item[$B$] Buoyancy force magnitude, $B=B_{mass}g$ (N).
\item[$B_{mass}$] Buoyancy equivalent mass (kg).
\item[$BG$] Distance between center of buoyancy and center of gravity (m).
\item[$C_m$] Pitch-moment coefficient about $+y_b$ (positive nose-up) (--).
\item[$C_X$] Force coefficient along $+x_b$ (--).
\item[$C_Z$] Force coefficient along $+z_b$ (--).
\item[$C_{cable,q}$] Cable damping coefficient in pitch (N$\cdot$m$\cdot$s/rad).
\item[$d_q$] Linear rotational damping coefficient (N$\cdot$m$\cdot$s/rad).
\item[$d_{q,abs}$] Quadratic rotational damping coefficient (N$\cdot$m$\cdot$s$^2$/rad$^2$).
\item[$g$] Gravitational acceleration (m/s$^2$).
\item[$I_y$] Effective pitch inertia used in the ODE denominator (kg$\cdot$m$^2$).
\item[$I_{yy}$] Rigid-body pitch moment of inertia (kg$\cdot$m$^2$).
\item[$I_{water,inner}$] Inner-water equivalent pitch inertia (kg$\cdot$m$^2$).
\item[$K_{cable}$] Cable torsional stiffness (N$\cdot$m/rad).
\item[$L_{ref}$] Reference length used in moment coefficient scaling (m).
\item[$m_{dry}$] Dry mass of the rigid platform body (kg).
\item[$m_{water,inner}$] Inner-water equivalent mass (kg).
\item[$m_{wet}$] Wet mass (dry mass plus inner-water equivalent mass) (kg).
\item[$m_x$] Effective surge inertia used in the ODE denominator (kg).
\item[$m_z$] Effective heave inertia used in the ODE denominator (kg).
\item[$M_{bg}$] Buoyancy restoring moment (N$\cdot$m).
\item[$M_{cable}$] Cable-induced restoring/damping moment (N$\cdot$m).
\item[$M_{cfd}$] CFD-derived hydrodynamic pitching moment (N$\cdot$m).
\item[$M_{damp}$] Rotational damping moment (N$\cdot$m).
\item[$M_{thruster}$] Thruster-induced pitching moment (N$\cdot$m).
\item[$q$] Pitch rate, $q=\dot{\theta}$ (rad/s).
\item[$Q$] Dynamic pressure, $Q=\tfrac{1}{2}\rho V^2$ (Pa).
\item[$T$] Thrust force along $+x_b$ (N).
\item[$u$] Surge velocity along $+x_b$ (m/s).
\item[$V$] Body-frame speed magnitude, $V=\sqrt{u^2+w^2}$ (m/s).
\item[$w$] Heave velocity along $+z_b$ (m/s).
\item[$W$] Weight magnitude, $W=m_{dry}g$ (N).
\item[$x_b$] Forward offset of buoyancy center in the body frame (m).
\item[$X_{cfd}$] CFD-derived hydrodynamic force along $+x_b$ (N).
\item[$X_{\dot{u},total}$] Total surge added-mass derivative (kg).
\item[$X_{cp}$] Pressure-center location used for mechanism interpretation (m).
\item[$z_b$] Downward offset of buoyancy center in the body frame (m).
\item[$Z_{cfd}$] CFD-derived hydrodynamic force along $+z_b$ (N).
\item[$Z_{\dot{w},total}$] Total heave added-mass derivative (kg).
\end{description}

\subsection*{Greek Symbols}
\begin{description}
\item[$\alpha$] Angle of attack, $\alpha=\mathrm{atan2}(w,u)$ (deg).
\item[$\eta$] Configuration-state vector, $\eta=[\theta]$ (--).
\item[$\theta$] Pitch angle (positive nose-up) (rad).
\item[$\theta_0$] Initial pitch angle of a free-decay segment (deg).
\item[$\theta_{eq}$] Cable equilibrium angle (rad).
\item[$\mu_x$] Surge-channel permeability coupling parameter (--).
\item[$\mu_z$] Heave-channel permeability coupling parameter (--).
\item[$\mu_{\theta}$] Pitch-channel permeability coupling parameter (--).
\item[$\nu$] Velocity-state vector, $\nu=[u,w,q]^T$ (--).
\item[$\rho$] Water density (kg/m$^3$).
\end{description}

\subsection*{Subscripts and Superscripts}
\begin{description}
\item[$(\cdot)_b$] Quantity expressed in the body frame.
\item[$(\cdot)_{inner}$] Inner-water contribution.
\item[$(\cdot)_{outer}$] Outer-fluid contribution.
\item[$(\cdot)_{total}$] Combined outer-fluid and permeability-corrected inner contribution.
\item[$(\cdot)_0$] Initial value at segment start.
\end{description}

\subsection*{Abbreviations}
\begin{description}
\item[AoA] Angle of attack.
\item[CFD] Computational fluid dynamics.
\item[CI] Confidence interval.
\item[CV] Cross-validation.
\item[DOF] Degree of freedom.
\item[MAE] Mean absolute error.
\item[NNLS] Non-negative least squares.
\item[RMSE] Root mean square error.
\item[TAAE] Time-averaged absolute error.
\item[TASE] Time-averaged signed error.
\end{description}
